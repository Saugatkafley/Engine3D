61 Vertex vertexIntersectPlane(vec3& point_plane, vec3& normal_plane, Vertex& lineStart, Vertex& lineEnd, float& t)
62 {
63 Vertex vo;
64 float plane_D = -normal_plane.dot(point_plane);
65 float ad = lineStart.position.dot(math::toVec4(normal_plane));
66 float bd = lineEnd.position.dot(math::toVec4(normal_plane));
67 t = (-plane_D - ad) / (bd - ad);
68 vo.position = lineStart.position + (lineEnd.position - lineStart.position) * t;
69 vo.normal = lineStart.normal + (lineEnd.normal - lineStart.normal) * t;
70 vo.tex = lineStart.tex + (lineEnd.tex - lineStart.tex) * t;
71 return vo;
72 }

74 int triClipPlane(vec3 point_plane, vec3 normal_plane, Triangle& t1, Triangle& t2)
75 {
76 // Returns signed shortest-distance from point to plane
77 auto distance = [&](vec3 p)
78 {
79 return (normal_plane.dot(p) - normal_plane.dot(point_plane));
80 };
81
82 Vertex* insideVertex[3]{}; int numInsidePoints = 0;
83 Vertex* outsideVertex[3]{}; int numOutsidePoints = 0;
84
85 // distance of each vertex to plane
86 float d0 = distance(math::toVec3(this->vertex[0].position));
87 float d1 = distance(math::toVec3(this->vertex[1].position));
88 float d2 = distance(math::toVec3(this->vertex[2].position));
89
90 if (d0 >= 0) { insideVertex[numInsidePoints++] = &this->vertex[0]; }
91 else { outsideVertex[numOutsidePoints++] = &this->vertex[0]; }
92
93 if (d1 >= 0) { insideVertex[numInsidePoints++] = &this->vertex[1]; }
94 else { outsideVertex[numOutsidePoints++] = &this->vertex[1]; }
95
96 if (d2 >= 0) { insideVertex[numInsidePoints++] = &this->vertex[2]; }
97 else { outsideVertex[numOutsidePoints++] = &this->vertex[2]; }
98
99 if (numInsidePoints == 0)
100 {
101 return 0;
102 }
103 if (numInsidePoints == 3)
104 {
105 t1 = *this;
106 return 1;
107 }
108 if (numInsidePoints == 1 && numOutsidePoints == 2)
109 {
110 t1 = *this;
111
112 t1.vertex[0] = *insideVertex[0];
113
114 float t{};
115 t1.vertex[1] = vertexIntersectPlane(point_plane, normal_plane, *insideVertex[0], *outsideVertex[0], t);
116 t1.vertex[2] = vertexIntersectPlane(point_plane, normal_plane, *insideVertex[0], *outsideVertex[1], t);
117
118 return 1;
119 }
120 if (numInsidePoints == 2 && numOutsidePoints == 1)
121 {
122 t1 = *this;
123
124 t1.vertex[0] = *insideVertex[0];
125 t1.vertex[1] = *insideVertex[1];
126 t2.vertex[0] = *insideVertex[1];
127
128 float t{};
129 t1.vertex[2] = vertexIntersectPlane(point_plane, normal_plane, *insideVertex[0], *outsideVertex[0], t);
130 t2.vertex[1] = t1.vertex[2];
131 t2.vertex[2] = vertexIntersectPlane(point_plane, normal_plane, *insideVertex[1], *outsideVertex[0], t);
132
133 return 2;
134 }
